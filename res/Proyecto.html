# Calcula los cuantiles de interes de average_rating
cuantiles <- quantile(base_copia$average_rating, probs = c(0.33, 0.66))

# Crea una columna categorica a base del rating promedio
base_copia <- base_copia %>%
  mutate(rating_category = case_when(
    average_rating < cuantiles[1] ~ "Bajo",
    average_rating >= cuantiles[1] & average_rating < cuantiles[2] ~ "Medio",
    average_rating >= cuantiles[2] ~ "Alto"
  ))

# Obtiene los 10 autores más populares
top_authors <- df_resultado[[1]] # asumo que la primera columna es el autor

# Crea una lista para almacenar los resultados
tabla_contingencia <- data.frame(
  Autor = character(),
  Bajo = integer(),
  Medio = integer(),
  Alto = integer(),
  stringsAsFactors = FALSE
)

# Para cada autor cuenta el numero de observaciones en cada una de las clases nuevas
for (autor in top_authors) {
  # Filtra las publicaciones donde el autor aparezca en la lista de autores
  publicaciones_autor <- base_copia %>%
    filter(str_detect(Author, fixed(autor)))
  
  # Cuenta las observaciones
  conteos <- publicaciones_autor %>%
    count(rating_category) %>%
    pivot_wider(names_from = rating_category, values_from = n, values_fill = 0)
  
  # Se asegura de que las columnas estén completas
  if (!"Bajo" %in% names(conteos)) conteos$Bajo <- 0
  if (!"Medio" %in% names(conteos)) conteos$Medio <- 0
  if (!"Alto" %in% names(conteos)) conteos$Alto <- 0
  
  # Crea el data frame final
  tabla_contingencia <- rbind(
    tabla_contingencia,
    data.frame(
      Autor = autor,
      Bajo = conteos$Bajo,
      Medio = conteos$Medio,
      Alto = conteos$Alto
    )
  )
}
# Extrae solo las columnas de frecuencias (sin la columna Autor)
tabla_chi <- tabla_contingencia %>%
  dplyr::select(Bajo, Medio, Alto) %>%
  as.matrix()

# Realiza la prueba de independencia chi-cuadrado 
prueba_chi <- chisq.test(tabla_chi)

# Mouestra los resultados
print(prueba_chi)
## 
##  Pearson's Chi-squared test
## 
## data:  tabla_chi
## X-squared = 22602, df = 17350, p-value < 2.2e-16
